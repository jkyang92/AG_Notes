\section{Lecture 6 (Andreas Hochenegger):  Bondal-Thomsen}
The Bondal-Thomsen collection is a natural set of line bundles on a toric variety that can be characterized in several different ways.  The following definition is originally due to Bondal \cite{Bondal}.  Consider the short exact sequence

\begin{equation}\label{eqn:SESCl}
0\to M\overset{\alpha}{\longrightarrow} \ZZ^n \overset{\deg}{\longrightarrow} \Cl(X) \to 0.
\end{equation}
We write $\alpha_\rho$ for the projection of $\alpha$ onto the basis vector corresponding to $\rho$ in $\Sigma(1)$; expressed in terms of the pairing between $M$ and $N$ this is $\alpha_\rho(m) = \langle m, -u_\rho\rangle$. By abuse of notation, we again write $\alpha_{\rho}$ for the natural extension of $\alpha_{\rho}$ to a map
$M\otimes \QQ\to \QQ^n$.
%Thus,
%Identifying $\ZZ^n$ with the set of torus invariant divisors, $\alpha(m) = \sum_{\rho} \alpha_\rho(m) D_\rho$.  and the class of $\alpha(m)$ is $0$ in $\Cl(X)$ for all $m\in M$.

Further, 
we define $\lfloor \alpha \rfloor\colon M\otimes \QQ \to \ZZ^n$ by the formula:
\[
  \lfloor \alpha \rfloor(m) :=  \sum_{\rho} \lfloor \alpha_\rho(m) \rfloor D_\rho.
\]
Note that though $\alpha(m)$ maps to the trivial element of $\Cl(X)\otimes \QQ$, this may not be the case for $\lfloor \alpha \rfloor(m)$.

\begin{defn}\label{defn:BTcollection}
    The \defi{Bondal-Thomsen collection} $\Theta_X\subseteq \Cl(X)\otimes \QQ$ is the image of $\deg \circ \lfloor \alpha \rfloor$.
    Elements of $\Theta_{X}$ are called Bondal-Thomsen divisor classes.
 \end{defn}
 
Note that $\Theta_{X}$ actually depends only on the  Cox ring $k[\NN^{n}]$ with its grading,
not on the toric variety $X$.

If $m\in M$, then $\alph(m) = \alpha(m)$, and this  maps to $0$ in $\Cl(X)$, so we may
consider $\Theta_X$ as the image of the induced map $(M\otimes \QQ) / M \to \Cl(X)\otimes \QQ$ which we also
call $\alph$. (This is where we need to regard the Bondal-Thomsen classes as living in $\Cl(X)\otimes \QQ$.

\begin{example} Consider the case $X=\P^{2}$. Then $M=\ZZ^2$ and $\ZZ^{n} = \ZZ^{3} =  \oplus_{i=0}^{2} \ZZ D_{i}$, with $\deg(D_{i}) = 1 \in \Cl(\PP^{2}) = \ZZ$
for each $i$. 

The map $\alpha$ is represented by the matrix
    \[
    \begin{pmatrix}
        -1&0\\ 0&-1\\1&1
    \end{pmatrix} .
    \]   
    Thus
        \[
    \lfloor \alpha \rfloor (a,b) = \lfloor -a\rfloor D_0 + \lfloor -b\rfloor D_1 + \lfloor a+b \rfloor D_2,
    \]
    and as a map to $\Cl(\PP^{2}) = \ZZ$ this becomes 
      \[
    \alph(a,b) =\lfloor -a\rfloor + \lfloor -b\rfloor + \lfloor a+b\rfloor .
    \]
    Any point in $(M\otimes \QQ)/ M$, can be represented uniquely by an element $(a,b)\in M\otimes \QQ^2$ with $0\leq a,b <1$.  Thus   the Bondal-Thomsen collection is the set
           \[
    \alph(a,b) =\lfloor -a\rfloor + \lfloor -b\rfloor + \lfloor a+b\rfloor \text{ with } (a,b) \in [0,1)^{2}
    \]

This is easy to compute and with $(a,b) \in [0,1)^{2}$ we have
$$
\alph(a,b) = \begin{cases}
0 & \text{if } a=b=0\\
-2& \text{if both } a,b \text{ are nonzero but } a+b<1\\
-1 & \text{otherwise.}
\end{cases}.
$$
Thus $\Theta_{\P^{2}} = \{-2, -1, 0\}$.
\end{example}




we can think of the Bondal-Thomsen classes as lattice points in the image
under the degree map of the product
of $(-1,0]^{n}\subset \RR^{n}$; such a convex set is called a \emph{zonotope}. 

\begin{defn}\label{defn:KoszulZonotope}
    The \defi{Bondal zonotope} is the locus
    \[
    \mathcal Z\ce\sum_{\rho} a_\rho \deg(x_\rho) \subseteq  \Cl(X)\otimes \RR \text{ with } a_\rho \in (-1,0].
    \]
\end{defn}

\begin{example}
    The Bondal zonotope $\PP^2$   is  the half-open interval $(-3,0] \subset \RR = \Cl(\P^{2})\otimes \RR$.\end{example}
    
\begin{thm}
    The following conditions on an element $d\in \Cl(X)$ are equivalent:.
    \begin{enumerate}
        \item  $d$ is a Bondal-Thomsen class.
        \item  $d$ is equivalent to a rational divisor of the form $\sum_{\rho} a_\rho D_\rho$ with $a_\rho \in (-1,0].$
        \item  $d\otimes 1 \in \Cl(X) \otimes \RR$  lies in the Bondal zonotope of $X$.
        \item  In case $X$ is smooth: $d$ represents a line bundle that appears as a direct summand of
        $F_{\ell,*}\cO_X$ for some $\ell$, or, equivalently, for every sufficiently large $\ell$ \footnote{According to \cite[Remark 5.4]{HHL} this can be done in the simplicial case
        by using the toric stack in place of the variety.}.
     \end{enumerate}
\end{thm}
\begin{proof}
(1) $\equiv$ (2):
As explained after Definition~\ref{defn:BTcollection}, the Bondal-Thomsen collection is the image of $M\otimes Q)/M$ under the
map $\alph$, and every element of $M\otimes Q)/M$ is represented by an element with coefficients in $(-1,0].$

(2) $\equiv$  (3): trivial.

(2) $\equiv$ (4) We follow the proof given in \cite[Theorem 1.1]{achinger}. We first show that 
$F_{\ell,*}\cO_X$ is a direct sum of line bundles. Since
 $X$ has an open
affine covering by affine spaces it suffices to treat the case $X = \AA^{d}$, and in this case 
$F_{\ell,*}\cO_X$ is the polynomial ring $k[x_{1},...,x_{n}]$ regarded as a module over
$S := k[x_{1}^{\ell},\dots,x_{n}^{\ell}$, which is obviously a direct sum of the rank 1 free modules
$x^{a}S$ for all multi-indices $a$ such with entries $0\leq a_{i}<\ell$. 

To see that these patch together,
note that for any sheaf $L$ on $X$ the sheaf $F_{\ell,*}\cO_X$ carries a natural action by the
group $G := \NN^{n}/\ell \NN^{n} \cong \ZZ/\ell^{n}$, acting by a character of $G$.
In the case of $\AA^{n}$ the summand $S(a)$ is the eigensheaf for eigenvalue $a$ and it follows
that there is no problem patching the summands across different open affine sets.

More generally, $F_{\ell,*}\cO_X(b)$ corresponds, when restricted to an open set $\AA^{n}$,
to $k[x_{1},\dots,x_{n}](b)$ regarded as an $S$-module, and this decomposes in the same way,
but with a shift by $b$.

To prove that the Bondal-Thomsen sheaves appear in the decomposition
of $F_{\ell,*}\cO_X$
we define exponents $m(e,d)$, for $d,e\in \Cl(X)$, by the formula
$$
F_{\ell,*}\cO_X(d) = \oplus_{e} \cO_{X}(e)^{\oplus m(e,d)}.
$$
From the push-pull formula we get
$$\leqno{(1)} \qquad
F_{\ell,*} O_X(d)  \otimes \cO_{X}(-e)= F_{\ell,*}\cO_X(d-\ell e)
$$
\noindent from which we see that $m(e,d) = m(0,d-\ell e)$ depends only on the difference $d-\ell e$.
Thus we define $m(d) := m(0,d)$.

Assuming, for the moment, that $X$ is complete, so that $h^{0}(\cO_{X}(e))$ is finite-dimensional for all $e$, we 
consider the generating function
$$
h^{0}(F_{\ell,*}\cO_X(d)) := \sum_{e\in \Cl(X) }m(e,d) h^{0}(\cO_{X}(e)) = \sum_{e\in \Cl(X)}m(d-\ell E) h^{0}(\cO_{X}(e)).
$$
 Fixing a basis $\{\cO_{X}(d_{i})\}_{i = 1,\dots,\rho}$ of $\Cl(X)$ we define generating functions
 $$
 s(t) = \sum_{a \in \ZZ^{\rho}} h^{0}( \cO_{X}(\sum_{i}a_{i}d_{i})) t^{a}, \qquad
 m(t) = \sum_{a \in \ZZ^{\rho}} m(\cO_{X}(\sum_{i}a_{i}d_{i})) t^{a}.
 $$
 where as usual $a = (a_{1}\dots,a_{\rho})$ is a multi-index.
From equation (1) we get
$$
s(t) = m(t)s(t_{1}^{\ell a_{1}}\cdots t_{\rho}^{\ell a_{\rho}}).
$$

Now $h^{0}(\cO_X(d))$ is the number of degrees $d' \in \text{Div}(X) = \ZZ^{n}$ with 
$\cO_X(d') \cong \cO_X(d)$, that is, the number of $d' \in \ZZ^{n}$ that map to a given $d = \deg(d')\in \Cl(X)$,
so
$$
s(t) = \sum_{d'\in ZZ^{n}} t^{\deg d'} = \prod_{i = 1}^{n}\frac{1}{1-t^{\deg e_{i}}}
$$
where $e_{i}$ is the $i$-th basis element of $\ZZ^{n}$. Thus
$$
m(t) = 
\frac{s(t)}{s((t_{1}^{\ell a_{1}}}\cdots t_{\rho}^{\ell a_{\rho}}) = 
\prod_{i=1}^{n}\frac{1-\ell \deg e_{i})}{1 -t^{\deg e_{i}}} = 
\prod_{i=1}^{n}(1+t^{\deg e_{i}})+\cdots + t^{(\ell-1)\deg e_{i}}).
$$
Comparing coefficients, it follows that 
$m(d)$ is the number of divisors $d'$ with $\deg(d') = d\in \Cl(X)$ that have the form
 $d' = \sum a_{i}e_{i}$ with $0\leq a_{i}\leq \ell$.

One important fact about the Bondal-Thomsen collection is the following easy consequence of Demazure Vanishing.\footnote{A similar statement, proven via similar methods, holds for reasonable toric stacks; see \cite{king}*{Lemma 2.20} for the precise statement.}
%\michael{This is (close to) \cite{king}*{Lemma 2.20}, which isn't that easy, right?}
\begin{thm}[Demazure Vanishing] \label{thm:demazure}
If $X$ is a simplicial, projective toric variety and $D$ is a nef $\QQ$-Cartier divisor on $X$, then
$H^p(X, \cO_{X}(\lfloor D \rfloor )) = 0$
for all $p > 0$.
\end{thm}
For a proof, see~\cite[Theorem 9.2.3]{CLS}.

\begin{prop}\label{prop:Demauzre}
    Let $a\in \Cl(X)$ be a nef degree and $-d\in \Theta_X$ be a Bondal--Thomsen degree.  We have
    $
    H^p(X,\cO_X(a-d)) = 0 \text{ for all } p>0.
    $
\end{prop}
\begin{proof}
    Since $a-d$ is an element of the class group, we can choose an integral toric divisor $B=\sum_{i} \alpha_iD_i$, with $\alpha_i\in \ZZ$, corresponding to $a-d$.  Since $-d$ is a Bondal--Thomsen degree, we can choose a toric divisor $D=\sum_i \epsilon_iD_i$, where $\epsilon_i \in [0,1)$, corresponding to $d$.

    Consider the toric $\QQ$-divisor $B+D$.  This is nef, because its class is $(a-d)+d=a$.  And we have $\rfloor B+D\lfloor = B$ which has class $a-d$.  Theorem~\ref{thm:demazure} thus implies $ H^p(X, \cO_{X}(\lfloor B+D \rfloor ))=0$ for all $p>0$.  And we have:
    \[
    H^p(X, \cO_{X}(\lfloor B+D \rfloor )) =  H^p(X, \cO_{X}(B)) =  H^p(X, \cO_{X}(a-d)).
    \]
\end{proof}

\begin{cor}\label{cor:BTNoCohom}
    If $-d\ \in \Theta_X$ is a Bondal--Thomsen degree, then $H^i(X,\cO_X(-d))=0$ for all $i>0$.  If $d\ne 0$, then $H^0(X,\cO_X(-d))$ also vanishes.
\end{cor}
\begin{proof}
    For $p>0$, we can apply Proposition~\ref{prop:Demauzre} with $a=0$.  For $H^0$, we note that when $d\ne 0$, $\cO_X(-d)$ is anti-effective and so has no global sections.
\end{proof}
