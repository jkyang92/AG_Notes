We now turn from projective space to the wider world of toric varieties, whose combinatorial description makes them an ideal candidate for scruitinizing the structure of derived categories. Throughout, $N\cong\ZZ^d$ is the lattice of \defi{one-parameter subgroups} of a torus $T = N\otimes_\ZZ \kk^*\cong(\kk^*)^d$ and its dual $M=\Hom_\ZZ(N,\ZZ)$ is the \defi{character lattice} of $T$. The usual, somewhat cryptic incantation is: a \defi{toric variety} is a normal variety $X$ containing $T$ as a dense open subset, together with an action of $T$ on $X$ extending its action on itself. The next two lectures focus on their combinatorial and algebraic structure instead. Our references are \cite[Ch.~3--5]{CLS} and David Cox's paper \cite{cox}.

\subsection{Toric varieties from fans}

An \defi{affine} toric variety is built from a cone. Let $\sigma\subseteq N\otimes\RR$ be a strongly convex rational polyhedral cone, with dual cone $\sigma^\vee\subseteq M\otimes\RR$. Then $S_\sigma=\sigma^\vee\cap M$ is a finitely generated semigroup, and $U_\sigma=\Spec \kk[S_\sigma]$ is an affine toric variety. Using these affine pieces, an arbitrary normal toric variety is glued together along the combinatorics of a \defi{fan}.

\begin{defn}
  A \defi{fan} $\Sigma$ in $N\otimes\RR$ is a finite collection of strongly convex rational polyhedral cones such that (i) every face of a cone in $\Sigma$ is again in $\Sigma$, and (ii) the intersection of any two cones in $\Sigma$ is a face of each. The associated toric variety $X_\Sigma$ is obtained by gluing the affine charts $\{U_\sigma\}_{\sigma\in\Sigma}$ along the charts of their common faces.
\end{defn}

The dictionary between the geometry of $X_\Sigma$ and combinatorics of $\Sigma$ is explained in \cite[\S3.1]{CLS}. In particular $X_\Sigma$ is \defi{smooth} (resp.\ \defi{simplicial}) if every cone of $\Sigma$ is generated by part of a $\ZZ$-basis (resp.\ an $\RR$-basis) of $N$, and it is \defi{complete} (equivalently proper) exactly when $\Sigma$ covers all of $N\otimes\RR$. We assume from now on that the rays of $\Sigma$ span $N\otimes\RR$, i.e.\ that $X_\Sigma$ has \defi{no torus factor}; this is automatic when $X_\Sigma$ is complete, as all of our examples are.

\subsection{Rays, divisors, and the class group}

Write $\Sigma(1)$ for the set of \defi{rays} (one-dimensional cones) of $\Sigma$, and let $u_\rho\in N$ denote the primitive generator of a ray $\rho$. By the Orbit--Cone Correspondence \cite[\S3.2]{CLS}, each ray $\rho$ corresponds to an irreducible $T$-invariant Weil divisor $D_\rho$ on $X_\Sigma$, and every $T$-invariant divisor is an integral combination $\sum_\rho a_\rho D_\rho$. Identifying $\Div_T(X_\Sigma)=\ZZ^{\Sigma(1)}$, there is an exact sequence \cite[Thm.~4.1.3]{CLS}
\begin{equation}\label{eqn:SEScl4}
  0\to M\xrightarrow{\ \alpha\ } \ZZ^{\Sigma(1)}\xrightarrow{\ \deg\ } \Cl(X_\Sigma)\to 0,
  \qquad \alpha(m)=\sum_\rho \langle m,u_\rho\rangle D_\rho,
\end{equation}
where $\Cl(X_\Sigma)$ is the \defi{divisor class group}. The map $\alpha$ gives the principal divisor of the character $\chi^m$, and $\deg$ sends a divisor to its class; injectivity of $\alpha$ uses our assumption that the rays span $N\otimes\RR$.

\begin{remark}[Cartier and Weil divisors]\label{rem:PicCl}
  Inside the class group sits the Picard group $\Pic(X_\Sigma) \xhookrightarrow{} \Cl(X_\Sigma)$ of \emph{Cartier} divisor classes, and the gap between the two measures the singularities of $X_\Sigma$. The toric variety is smooth if and only if $\Pic(X_\Sigma) = \Cl(X_\Sigma)$ (note that isomorphism is not sufficient), and it is simplicial if and only if $\Pic(X_\Sigma)$ has finite index in $\Cl(X_\Sigma)$, in which case every Weil divisor is \defi{$\QQ$-Cartier}, meaning some positive multiple is Cartier \cite[Prop.~4.2.6-7]{CLS}. Example~\ref{ex:P112charts} below illustrates both phenomena on $\PP(1,1,2)$; the distinction is also apparent in the module--sheaf correspondence and in Demazure vanishing in Lecture~\ref{lec:5-toric-cox}.
\end{remark}

\subsection{The Cox ring and irrelevant ideal}

Assign to each ray $\rho\in\Sigma(1)$ a variable $x_\rho$ and consider
\[ S = \Cox(X_\Sigma) = \kk[x_\rho : \rho\in\Sigma(1)] \]
to be a polynomial ring graded by $\Cl(X_\Sigma)$ via $\deg(x_\rho)=[D_\rho]$. This is the \defi{Cox ring} (or \defi{homogeneous coordinate ring}) of $X_\Sigma$. A monomial $x^{\mathbf a}=\prod_\rho x_\rho^{a_\rho}$ has degree $[\sum_\rho a_\rho D_\rho]\in\Cl(X_\Sigma)$.

The geometry of $X_\Sigma$, encoded in the combinatorics of the maximal cones of $\Sigma$, is recorded by the so-called \defi{irrelevant ideal}
\[
B = \bigl\langle\, x^{\widehat\sigma}\ :\ \sigma\in\Sigma_{\max}\,\bigr\rangle, \qquad
x^{\widehat\sigma} = \prod_{\rho\;\notin\;\sigma(1)} x_\rho,
\]
which is a squarefree monomial ideal generated, for each maximal cone $\sigma$, by the product of the variables whose rays are not in $\sigma$.

\begin{remark}[Primitive collections]\label{rem:primitive}
  The locus $V(B)\subseteq\bA^{\Sigma(1)}$ cut out by the irrelevant ideal is a union of coordinate subspaces, indexed by the \defi{primitive collections} of $\Sigma$: the minimal subsets $\mathcal C\subseteq\Sigma(1)$ that do not span a cone of $\Sigma$, although every proper subset does. Explicitly,
  \[ V(B) = \bigcup_{\mathcal C} V(x_\rho : \rho\in\mathcal C) \quad \text{\cite[Prop.~5.1.6]{CLS}}. \]
  Observe that the ring $S$, its grading, and the group $G$ below depend only on the rays $\Sigma(1)$, whereas $B$ and $V(B)$ remember the maximal cones; in particular, different fans with the same rays share the graded ring $S$ but not the ideal $B$. We return to this observation in Lecture~\ref{lec:5-toric-cox} and, systematically, in Lecture~\ref{lec:11-secondary-fan}.
\end{remark}

\begin{thm}[quotient construction, \cite{CLS}*{Thm.~5.1.11}]\label{thm:coxquotient}
  Let $X_\Sigma$ be a toric variety with no torus factor and let $G=\Hom_\ZZ(\Cl(X_\Sigma),\kk^*)$ act on $\Spec S=\bA^{\Sigma(1)}$ via the grading on $S$. Then
  \[ X_\Sigma \cong \bigl(\Spec S\setminus V(B)\bigr)\,/\!\!/\,G \]
  is a good categorical quotient, and it is a geometric quotient if and only if $X_\Sigma$ is simplicial.
\end{thm}

Thus a point of a simplicial $X_\Sigma$ is a $G$-orbit of ``homogeneous coordinates'' $[a_\rho]_{\rho\in\Sigma(1)}$ not lying in the excluded locus $V(B)$. For $\PP^n$ this recovers the familiar $\PP^n=(\bA^{n+1}\setminus\{0\})\,/\,\kk^*$.

\begin{example}\label{ex:toricrunning}
  We now illustrate several toric surfaces as running examples.
  \begin{enumerate}
  \item \textbf{Projective space $\PP^n$.}  The fan has as rays the standard basis $u_1,\dots,u_n$ and their complement $u_0 = -u_1-\dots-u_n$, with maximal cones spanned by all $n$-subsets. Then $S = \kk[x_0,\dots,x_n]$ with the standard $\ZZ$-grading $\deg(x_i) = 1$ and $B = (x_0,\dots,x_n)$: the only primitive collection is all of $\Sigma(1)$, so $V(B) = \{0\}$.

  \item \textbf{Products $\PP^{n_1}\times\dots\times\PP^{n_r}$.}  The Cox ring is a multigraded polynomial ring with $\ZZ^r$-grading and $B = \bigcap_i (x_{i,0},\dots,x_{i,n_i})$; the primitive collections are the $k$ sets of rays of the factors.

  \item \textbf{Weighted projective space $\PP(a_0,\dots,a_n)$, where $\gcd(a_0,\dots,a_n) = 1$.}  The fan is given by the quotient lattice $N = \ZZ^{n+1}/(a_0,\dots,a_n)$, the rays being the images of the standard basis vectors. Then, $S = \kk[x_0,\dots,x_n]$ and $B = (x_0,\dots,x_n)$ with $\deg(x_i) = a_i$. Note that the inclusion of the Picard group in the class group is $\on{lcm}(a_0,\dots,a_n)\ZZ \xhookrightarrow{} \ZZ$, thus weighted projective spaces are generally singular; see Example~\ref{ex:P112charts}.

  \item \textbf{Hirzebruch surfaces $\mathcal H_a$ ($a\ge 1$).}  The surface $\mathcal H_a=\PP(\cO_{\PP^1}\oplus\cO_{\PP^1}(a))$ is the toric variety of the fan in $N=\ZZ^2$ with rays
    \[
    u_0=e_1,\qquad u_1=e_2,\qquad u_2=-e_1+ae_2,\qquad u_3=-e_2,
    \]
    and the four maximal cones spanned by neighbouring pairs (Figure~\ref{fig:fans}). Writing $x_i$ for the variable of $u_i$, the sequence \eqref{eqn:SEScl4} gives the relations $D_0\sim D_2$ and $D_3\sim D_1+aD_2$, so in the basis $([D_2],[D_3])$ of $\Cl(\mathcal H_a)=\ZZ^2$ the degrees are
    \[
    \deg(x_0)=\deg(x_2)=(1,0),\qquad \deg(x_1)=(-a,1),\qquad \deg(x_3)=(0,1),
    \]
    and the primitive collections $\{u_0,u_2\}$ and $\{u_1,u_3\}$ give $B=(x_0,x_2)\cap(x_1,x_3)$. Geometrically, $(1,0)$ is the class of a fiber of $\mathcal H_a\to\PP^1$ and class $(-a,1)$ is the class of the divisor with self-intersection $-a$.
  \end{enumerate}
\end{example}

\begin{example}[$\PP(1,1,2)$ in detail]\label{ex:P112charts}
  Let $S = \kk[x,y,z]$ with $\deg(x)=\deg(y)=1$ and $\deg(z)=2$ and $B = (x,y,z)$. By Theorem~\ref{thm:coxquotient}, $\PP(1,1,2) = (\bA^3\setminus\{0\})\,/\,\kk^*$ for the action $\lambda\cdot(x,y,z)=(\lambda x,\lambda y,\lambda^2 z)$. The irrelevant ideal gives the open affine cover $D_+(x)\cup D_+(y)\cup D_+(z)$, where
  \[ D_+(x) = \Spec\bigl(S[x^{-1}]\bigr)_0 = \Spec \kk\Bigl[\tfrac{y}{x},\tfrac{z}{x^2}\Bigr]\cong\bA^2, \]
  and analogously for $D_+(y)\cong\bA^2$, but
  \[ D_+(z) = \Spec\bigl(S[z^{-1}]\bigr)_0 = \Spec\kk\Bigl[\tfrac{x^2}{z},\tfrac{xy}{z},\tfrac{y^2}{z}\Bigr]\cong \Spec \kk[u,v,w]/(uw-v^2) \]
  is the quadric cone. Thus $\PP(1,1,2)$ is singular exactly at the point $[0:0:1]$, where the quotient has stabilizer $\mu_2 = \{\pm1\}$. In particular, a divisor class $d\in\Cl(\PP(1,1,2)) = \ZZ$ is Cartier if and only if $d$ is a multiple of $2 = \on{lcm}(1,1,2)$, so $\Pic(\PP(1,1,2)) = 2\ZZ \subsetneq \ZZ$, as Remark~\ref{rem:PicCl} predicts for a simplicial but non-smooth toric variety. In particular the rank-one reflexive sheaf $\cO(1)$, corresponding to the module $S(1)$, is \emph{not} a line bundle. This point is further scruitinized in Lecture~\ref{lec:5-toric-cox}.
\end{example}
